Arrow Function (ES6)


  

    // Traditional function
function add(a, b) {
  return a + b;
}

// Arrow function
const add = (a, b) => a + b;
    // Traditional anonymous function	
function (a, b) {
  return a + b;
}	

// Arrow function
(a, b) => a + b;

Spread Operator (...)

Array spread [...arr], ES6 (2015) 
const colors = ['red', 'green'];
const moreColors = [...colors, 'blue'];

console.log(moreColors);
// ['red', 'green', 'blue']

Object spread { ...obj }, ES9 (2018) 
const person = { name: 'Tyler', age: 40 };
const updatedPerson = { ...person, age: 41 };

console.log(updatedPerson);
// { name: 'Tyler', age: 41 }
