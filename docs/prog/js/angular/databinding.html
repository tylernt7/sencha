Data Binding

Overview

Model: Property (Data)
View: Template (HTML)
Controller/Viewmodel: Component (element | component | directive)

One-way binding (Model to View)

HTML 
<!-- component.html -->
<h1>{{ title }}</h1> <!-- Interpolation -->
<h1 [innerText]="title"></h1> <!-- Property Binding -->
TypeScript 
// component.ts
export class AppComponent {
  title = 'Hello World';
}

One-way binding (View to Model)

HTML 
<!-- component.html -->
<button (click)="onSave()">Save</button> <!-- Event Binding -->
TypeScript 
// component.ts
export class AppComponent {
  onSave() {
    alert('Saved!');
  }
}

Two-way binding (View to Model to View)

HTML 
<!-- component.html --> 
<input [(ngModel)]="name" placeholder="Enter name"> <!-- Two-way Binding -->
<p>Hello {{ name }}</p>
TypeScript 
// component.ts
export class AppComponent {
  name: string = '';
}
